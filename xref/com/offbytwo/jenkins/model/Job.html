7   package com.offbytwo.jenkins.model;
8   
9   import com.google.common.base.Function;
10  import com.google.common.collect.Collections2;
11  import com.google.common.escape.Escaper;
12  import com.google.common.net.UrlEscapers;
13  
14  import java.io.IOException;
15  import java.util.Map;
16  
17  import static org.apache.commons.lang.StringUtils.join;
18  
19  public class Job extends BaseModel {
20  
21      private String name;
22      private String url;
23  
24      public Job() {}
25  
26      public Job(String name, String url) {
27          this();
28          this.name = name;
29          this.url = url;
30      }
31  
32      public String getName() {
33          return name;
34      }
35  
36      public String getUrl() {
37          return url;
38      }
39  
40      public JobWithDetails details() throws IOException {
41          return client.get(url, JobWithDetails.class);
42      }
43  
44      /**
45       * Trigger a build without parameters
46       */
47      public void build() throws IOException {
48          client.post(url + "build", true);
49      }
50  
51      public void build(boolean crumbFlag) throws IOException {
52          client.post(url + "build", crumbFlag);
53      }
54  
55      /**
56       * Trigger a parameterized build
57       *
58       * @param params the job parameters
59       * @throws IOException
60       */
61      public void build(Map<String, String> params) throws IOException {
62          String qs = join(Collections2.transform(params.entrySet(), new MapEntryToQueryStringPair()), "&");
63          client.post(url + "buildWithParameters?" + qs, null, null);
64      }
65  
66      @Override
67      public boolean equals(Object o) {
68          if (this == o) return true;
69          if (o == null || getClass() != o.getClass()) return false;
70  
71          Job job = (Job) o;
72  
73          if (name != null ? !name.equals(job.name) : job.name != null)
74              return false;
75          if (url != null ? !url.equals(job.url) : job.url != null) return false;
76  
77          return true;
78      }
79  
80      @Override
81      public int hashCode() {
82          int result = name != null ? name.hashCode() : 0;
83          result = 31 * result + (url != null ? url.hashCode() : 0);
84          return result;
85      }
86  
87      private static class MapEntryToQueryStringPair implements Function<Map.Entry<String, String>, String> {
88          @Override
89          public String apply(Map.Entry<String, String> entry) {
90              Escaper escaper = UrlEscapers.urlFormParameterEscaper();
91              return escaper.escape(entry.getKey()) + "=" + escaper.escape(entry.getValue());
92          }
93      }
94  }
