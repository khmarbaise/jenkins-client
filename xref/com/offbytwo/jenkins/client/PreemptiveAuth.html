7   package com.offbytwo.jenkins.client;
8   
9   import org.apache.http.HttpException;
10  import org.apache.http.HttpHost;
11  import org.apache.http.HttpRequest;
12  import org.apache.http.HttpRequestInterceptor;
13  import org.apache.http.auth.AuthScheme;
14  import org.apache.http.auth.AuthScope;
15  import org.apache.http.auth.AuthState;
16  import org.apache.http.auth.Credentials;
17  import org.apache.http.client.CredentialsProvider;
18  import org.apache.http.client.protocol.ClientContext;
19  import org.apache.http.protocol.ExecutionContext;
20  import org.apache.http.protocol.HttpContext;
21  
22  import java.io.IOException;
23  
24  class PreemptiveAuth implements HttpRequestInterceptor {
25  
26      @Override
27      public void process(HttpRequest request, HttpContext context) throws HttpException, IOException {
28          AuthState authState = (AuthState) context.getAttribute(ClientContext.TARGET_AUTH_STATE);
29  
30          if (authState.getAuthScheme() == null) {
31              AuthScheme authScheme = (AuthScheme) context.getAttribute("preemptive-auth");
32              CredentialsProvider credsProvider = (CredentialsProvider) context.getAttribute(ClientContext.CREDS_PROVIDER);
33              HttpHost targetHost = (HttpHost) context.getAttribute(ExecutionContext.HTTP_TARGET_HOST);
34              if (authScheme != null) {
35                  Credentials creds = credsProvider.getCredentials(
36                          new AuthScope(targetHost.getHostName(), targetHost.getPort()));
37                  if (creds == null) {
38                      throw new HttpException("No credentials for preemptive authentication");
39                  }
40                  authState.update(authScheme, creds);
41              }
42          }
43      }
44  }
