5   import javax.xml.bind.JAXBException;
6   import javax.xml.bind.Marshaller;
7   import java.io.StringWriter;
8   import java.util.ArrayList;
9   import java.util.List;
10  
11  import static org.junit.Assert.assertTrue;
12  
13  public class ParametersDefinitionPropertyTest extends BaseXmlMarshallingTest {
14  
15      @Test
16      public void shouldMarshallParametersDefinitionProperty() throws JAXBException {
17          // given
18          StringWriter sw = new StringWriter();
19          configureXmlMarshallerFor(ParametersDefinitionProperty.class);
20          marshaller.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, true);
21  
22          ParametersDefinitionProperty pdp = givenParametersDefinitionProperty();
23  
24          // when
25          marshaller.marshal(pdp, sw);
26  
27          // then
28          assertTrue(sw.toString().contains("<name>aaa</name>"));
29          assertTrue(sw.toString().contains("<description>aaa</description>"));
30          assertTrue(sw.toString().contains("<defaultValue>aaa</defaultValue>"));
31  
32          assertTrue(sw.toString().contains("<name>bbb</name>"));
33          assertTrue(sw.toString().contains("<description>bbb</description>"));
34          assertTrue(sw.toString().contains("<defaultValue>bbb</defaultValue>"));
35      }
36  
37      private ParametersDefinitionProperty givenParametersDefinitionProperty() {
38          StringParameterDefinition s1 = new StringParameterDefinition("aaa", "aaa", "aaa");
39          StringParameterDefinition s2 = new StringParameterDefinition("bbb", "bbb", "bbb");
40          List<StringParameterDefinition> list = new ArrayList<StringParameterDefinition>();
41          list.add(s1);
42          list.add(s2);
43          ParameterDefinitions pd = new ParameterDefinitions();
44          pd.setStringParams(list);
45          ParametersDefinitionProperty pdp = new ParametersDefinitionProperty();
46          pdp.setPd(pd);
47          return pdp;
48      }
49  }
